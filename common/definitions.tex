\newcommand{\area} [2] {
    \vspace*{-9pt}
    \begin{verse}
        \textbf{#1} #2
    \end{verse}
}

% \lineunder command
\newcommand{\lineunder} {
    \vspace{-8pt} \\
    \hspace{-18pt} \hrulefill \\
}

% \name command
% #1: Name
\newcommand{\name}[1] {
	{\Huge #1}
}

% \namesc command (same as \name but with small caps)
% #1: Name
\newcommand{\namesc}[1] {
	\name{\scshape{#1}}
}

% \photo command
% #1: Photo file path
\newcommand{\photo}[1] {
	\begin{adjustbox}{width=3.5cm, valign=m, keepaspectratio}
		\begin{tikzpicture}
			\clip (1.75cm, 1.75cm) circle(1.75cm);
			\includegraphics[width=3.5cm]{#1};
		\end{tikzpicture}
	\end{adjustbox}
}

% \header command
% #1: Header title
\newcommand{\header}[1] {
    \hspace{-18pt} \vspace{4pt} #1 \lineunder \vspace{4pt}
}

% \headersc command (same as \header but with small caps)
% #1: Header title
\newcommand{\headersc}[1] {
	\header{\textsc{#1}}
}

% \keyvalpair command
% #1: Key
% #2: Value
\newcommand{\keyvalpair}[2] {
	\textbf{#1:} & #2
}

\newcommand{\employer} [3] {

}
